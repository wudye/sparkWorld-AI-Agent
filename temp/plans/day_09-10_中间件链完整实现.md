### 任务9.1：中间件基类（1小时）

```python
# deerflow/agents/middlewares/base.py

from abc import ABC, abstractmethod
from deerflow.agents.thread_state import ThreadState

class BaseMiddleware(ABC):
    """中间件基类"""
    
    @abstractmethod
    def __call__(self, state: ThreadState) -> ThreadState:
        """执行中间件逻辑"""
        pass

class MiddlewareChain:
    """中间件链协调器"""
    
    def __init__(self, middlewares: list[BaseMiddleware]):
        self.middlewares = middlewares
    
    def __call__(self, state: ThreadState) -> ThreadState:
        """依次执行所有中间件"""
        for middleware in self.middlewares:
            try:
                state = middleware(state)
            except Exception as e:
                print(f"Middleware error: {e}")
                raise
        return state
```

```python
# deerflow/agents/middlewares/thread_data.py

import os
from deerflow.agents.middlewares.base import BaseMiddleware
from deerflow.agents.thread_state import ThreadState

class ThreadDataMiddleware(BaseMiddleware):
    """为每个thread创建隔离目录"""
    
    def __init__(self, base_path: str = "/tmp/deerflow"):
        self.base_path = base_path
    
    def __call__(self, state: ThreadState) -> ThreadState:
        # 从thread_data获取thread_id (需要从其他地方传入)
        thread_id = state.get('thread_id', 'default')
        
        # 创建目录
        workspace = os.path.join(self.base_path, thread_id, 'workspace')
        uploads = os.path.join(self.base_path, thread_id, 'uploads')
        outputs = os.path.join(self.base_path, thread_id, 'outputs')
        
        os.makedirs(workspace, exist_ok=True)
        os.makedirs(uploads, exist_ok=True)
        os.makedirs(outputs, exist_ok=True)
        
        # 更新state
        state['thread_data'] = {
            'workspace': workspace,
            'uploads': uploads,
            'outputs': outputs
        }
        
        return state
```

```python
from deerflow.agents.middlewares.base import BaseMiddleware
from deerflow.agents.thread_state import ThreadState

class TitleMiddleware(BaseMiddleware):
    """首次对话后自动生成标题"""
    
    def __call__(self, state: ThreadState) -> ThreadState:
        # 简单实现：第一条消息的前20个字符作为标题
        if len(state['messages']) == 1 and state['title'] is None:
            first_msg = state['messages'][0].content
            state['title'] = first_msg[:20] + "..."
        return state
```

## 📋 Day 10：中间件与Agent集成（6小时）

### 任务10.1：Agent节点设计（2小时）

```python
# deerflow/agents/lead_agent/agent.py (更新)

from langgraph.graph import StateGraph, START, END
from deerflow.agents.thread_state import ThreadState
from deerflow.agents.middlewares.base import MiddlewareChain
from deerflow.agents.middlewares.thread_data import ThreadDataMiddleware
from deerflow.agents.middlewares.sandbox import SandboxMiddleware
from deerflow.agents.middlewares.title import TitleMiddleware

def make_lead_agent(config):
    """创建主Agent"""
    
    # 1. 创建中间件链
    middlewares = [
        ThreadDataMiddleware('/tmp/deerflow'),
        SandboxMiddleware(),
        TitleMiddleware(),
    ]
    chain = MiddlewareChain(middlewares)
    
    # 2. 定义节点
    def agent_node(state: ThreadState) -> ThreadState:
        """Agent执行节点"""
        # 先执行中间件链
        state = chain(state)
        
        # 这里会调用LLM (后续实现)
        print(f"Agent processing with middleware-modified state")
        
        return state
    
    # 3. 创建StateGraph
    graph = StateGraph(ThreadState)
    graph.add_node("agent", agent_node)
    graph.add_edge(START, "agent")
    graph.add_edge("agent", END)
    
    return graph.compile()
```
